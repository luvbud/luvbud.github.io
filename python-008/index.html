3. 멤버변수

▷ 입력
# 멤버변수
class Unit:
    def __init__(self, name, hp, damage):
        # 멤버변수 = 클래스 내에서 정의된 변수
        self.name = name
        self.hp = hp
        self.damage = damage
        print("{0} 유닛이 생성되었습니다.".format(self.name))
        print("체력 {0}, 공격력 {1}".format(self.hp, self.damage))

# 레이스 : 공중 유닛, 비행기. 클로킹 (상대방에게 보이지 않음)
wraith1 = Unit("레이스", 80, 5)
print("유닛 이름 : {0}, 공격력 : {1}".format(wraith1.name, wraith1.damage))

# 마인드 컨트롤 : 상대방 유닛을 내 것으로 만드는 것 (빼앗음)
wraith2 = Unit("빼앗은 레이스", 80, 5)
wraith2.clocking = True # clocking 이라는 변수는 Unit 클래스 내에 없지만 클래스 외부에서 clocking 이라는 변수를 추가로 만들어 wraith2에 할당함
# 클래스 외부에서 내가 원하는 어떤 변수에 대해서 확장할 수 있음
# 확장된 변수는 내가 확장을 한 객체에 대해서만 적용이 됨
# 기존에 있던 다른 객체에 대해서는 적용이 안 됨

if wraith2.clocking == True:
    print("{0}는 현재 클로킹 상태입니다.".format(wraith2.name))

# if wraith1.clocking == True: # 적용 불가. 오류남.
#     print("{0}는 현재 클로킹 상태입니다.".format(wraith2.name))
출력
레이스 유닛이 생성되었습니다.
체력 80, 공격력 5
유닛 이름 : 레이스, 공격력 : 5
빼앗은 레이스 유닛이 생성되었습니다.
체력 80, 공격력 5
