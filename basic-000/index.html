# 1082. 16진수 구구단
n = int(input(), 16)
for i in range(1, 16):
    print("%X*%X=%X" % (n, i, n*i)) # %X : 16진수
출력
B

B*2=16
B*3=21
B*4=2C
B*5=37
B*6=42
B*7=4D

B*D=8F
B*E=9A
B*F=A5
